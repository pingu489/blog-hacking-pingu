Chuleta de Comandos Linux

# COMANDOS DE ARCHIVOS
ls # Listar directorios
ls -l # Listado detallado
ls -a # Listar todos los archivos incluyendo ocultos
cd /ruta/a/directorio # Cambiar de directorio
pwd # Mostrar el directorio actual
mkdir nombre_directorio # Crear un nuevo directorio
rmdir nombre_directorio # Eliminar un directorio vacío
rm nombre_archivo # Eliminar un archivo
rm -r nombre_directorio # Eliminar un directorio y su contenido recursivamente
touch nombre_archivo # Crear o actualizar un archivo
cat nombre_archivo # Mostrar el contenido de un archivo
more nombre_archivo # Ver contenido del archivo página a página
less nombre_archivo # Mejor visualización del contenido que 'more'
cp archivo_origen archivo_destino # Copiar archivos de origen a destino
mv nombre_antiguo nombre_nuevo # Renombrar o mover un archivo/directorio
ln -s archivo_destino nombre_link # Crear un enlace simbólico

# SSH (Secure Shell)
ssh usuario@host # Conectar al host como usuario
ssh -p puerto usuario@host # Conectar usando un puerto específico
ssh-keygen -t rsa # Generar clave RSA
ssh-copy-id usuario@host # Copiar tu clave al servidor remoto para login sin contraseña
scp archivo usuario@host:ruta # Copiar archivo al host remoto
scp usuario@host:/ruta/archivo /ruta/local # Copiar archivo desde host remoto

# BÚSQUEDA
grep patrón archivos # Buscar patrón en archivos
grep -r patrón directorio # Buscar recursivamente en directorio
find /directorio -name nombre* # Buscar archivos que empiezan con nombre en directorio
find /directorio -type f -name "*.txt" # Buscar archivos con extensión específica
locate nombre_archivo # Buscar archivo por nombre usando base de datos
updatedb # Actualizar base de datos usada por locate

# GESTIÓN DE PROCESOS
ps # Mostrar procesos activos
ps aux | grep nombre_proceso # Buscar proceso por nombre
top # Mostrar todos los procesos en ejecución
htop # Visor interactivo de procesos (requiere instalación)
kill pid # Matar proceso con pid dado
killall nombre_proceso # Matar todos los procesos con ese nombre
pkill nombre_proceso # Matar procesos por nombre
bg # Listar trabajos en segundo plano; reanudar trabajo parado en background
fg # Traer el trabajo más reciente al primer plano
jobs # Listar trabajos activos
nice -n prioridad comando # Ejecutar comando con prioridad específica
renice prioridad -p pid # Cambiar prioridad de un proceso en ejecución

# PERMISOS DE ARCHIVOS
chmod +x nombre_archivo # Hacer archivo ejecutable
chmod 755 nombre_archivo # Permitir lectura y ejecución al dueño y lectura a otros
chown usuario:grupo nombre_archivo # Cambiar dueño y grupo de un archivo
chgrp grupo nombre_archivo # Cambiar grupo propietario de un archivo
umask # Mostrar o establecer permisos por defecto para nuevos archivos
